1. docker exec -it namenode bash
2. hdfs dfs -mkdir -p /data/wordcount/input
3. docker cp data.txt namenode:data.txt
4. hdfs dfs -put /data.txt /data/wordcount/input/
5. docker cp mapper.py namenode:mapper.py
6. docker cp reducer.py namenode:reducer.py
7. chmod +x mapper.py reducer.py
8. hadoop jar /opt/hadoop-*/share/hadoop/tools/lib/hadoop-streaming-*.jar \
    -input /data/wordcount/input/data.txt \
    -output /data/wordcount/wc_output_v9 \
    -mapper "python mapper.py" \
    -reducer "python reducer.py"
